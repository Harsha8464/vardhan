public class FirstPrimeNumber {
    public static void main(String[] args) {
        int num = 2;
        boolean isPrime = true;
        
        while (true) {
            for (int i = 2; i <= Math.sqrt(num); i++) {
                if (num % i == 0) {
                    isPrime = false;
                    break;
                }
            }
            
            if (isPrime) {
                System.out.println("The first prime number is: " + num);
                break;
            }
            
            isPrime = true;
            num++;
        }
    }
}
